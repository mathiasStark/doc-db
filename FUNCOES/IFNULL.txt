-- FUNCOES
 
-- SAO BLOCOS DE CODIGOS QUE PODEM SER PROGRAMADOS, PARA RETORNAREM DADOS DO BANCO;

-- IFNULL
APLICADO EM UMA COLUNA PARA RETORNAR OUTRO VALOR SE O VALOR OBTIDO FOR NULO;
RECEBE COMO PARAMETRO UMA COLUNA E VERIFICA REGISTRO A REGISTRO SE EXISTE VALOR NULO, E OUTRO PARAMETRO STRING PARA SUBSTITUIR O NULO;

EX.:
	IFNULL(COLUNA, STRING-SUBSTITUI-NULL);

 SELECT 
	NOME, 
	IFNULL(EMAIL, 'NAO TEM EMAIL') AS EMAIL, 
	ESTADO, 
	NUMERO
 FROM CLIENTE C
 INNER JOIN ENDERECO e ON c.IDCLIENTE = e.ID_CLIENTE
 INNER JOIN TELEFONE t ON c.IDCLIENTE = t.ID_CLIENTE;
