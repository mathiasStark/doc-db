CREATE VIEW RELATORIO AS
 SELECT 
	c.NOME, 
	IFNULL(c.EMAIL, 'NAO TEM EMAIL') EMAIL, 
	e.ESTADO, 
	t.NUMERO
 FROM CLIENTE C
 INNER JOIN ENDERECO e ON c.IDCLIENTE = e.ID_CLIENTE
 INNER JOIN TELEFONE t ON c.IDCLIENTE = t.ID_CLIENTE;
