CHAVE ESTRANGEIRA
	
	CHAVE ESTRANGEIRA FK É A CHAVE PRIMARIA DE UMA TABELA QUE VAI ATÉ A OUTRA TABELA PARA FAZER REFERENCIA ENTRE REGISTROS.
	QUEM DEFINE A TABELA QUE VAI CONTER A FK É A CARDINALIDADE.
	QUEM DEFINE A CARDINALIDADE É A REGRA DE NEGÓCIOS.
	
	1X1
	EM RELACIONAMENTOS 1X1 A FK FICA NA TABELA MAIS FRACA. DEFINIDO PELO TIPO DE NEGOCIO.GERALMENTE O SUJEITO PRINCIPAL DO NEGOCIO É A TABELA MAIS FORTE.
	#UTILIZAR UNIQUE PARA RESTRINGIR DUPLICIDADE;
	
	1XN
	EM RELACIONAMENTOS 1XN A CHAVE ESTRANGEIRA FICARA SEMPRE NA CARDINALIDADE N.

EXEMPLO:
		
	NOME_COLUNA_FK TIPO,
	FOREIGN KEY(NOME_COLUNA_FK) REFERENCES TABELA_QUE_CONTEM_PK(PK)
	
	ID_CLIENTE INT,
	FOREIGN KEY(ID_CLIENTE) REFERENCES CLIENTE(IDCLIENTE)
