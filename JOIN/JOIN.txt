JOIN
	
	JOIN É A JUNCAO DE 2 OU MAIS TABELAS ATRTAVÉS DE DADOS QUE ESTÃO PRESENTES NESTAS TABELAS, GERALMENTE UTILIZAMOS A FK E A PK PARA COMPARAR ESTES VALORES;

EX.:
	SELECT COLUNAS_TABELA_ORIGEM_E_SECUNDARIA
	FROM TABELA_ORIGEM
	INNER JOIN TABELA_SECUNDARIA 
	ON PK_TABELA_ORIGEM = FK_TABELA_SECUNDARIA (CONDICAO PARA O JOIN)
	WHERE SELECAO;
