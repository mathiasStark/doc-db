CHAR E VARCHAR
	
	CADA CARACTER EQUIVALE A 1 BYTE.
	O CHAR NÃO VARIA UM CHAR(10) SEMPRE VALERA 10 BYTES, ENQUANTO UM VARCHAR(10) PODE OU NÃO TER TAMANHO 10 BYTES DEPENDENDO DO TIPO DE PALAVRAS QUE FOR INSERIDA.
	
	CHAR(10) = IZA TAMANHO 10 BYTES;
	VARCHAR(10) = IZA TAMANHO 3 BYTES;

	SEMPRE UTILIZAR CHAR PARA DADOS QUE TEM SEUS TAMANHOS PRE-DEFINIDOS E QUE NÃO SOFRERÃO ALTERAÇÃO.
	EXEMPLO SEXO, ESTADO, CPF ETC;
